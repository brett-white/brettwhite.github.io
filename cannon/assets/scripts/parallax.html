$(document).on('scroll', function () { var scrollPoint = $(window).scrollTop(); $('[parallax]').each(function () { var theElement = $(this); var factor = theElement.attr('parallax'); var offset = (theElement.attr('parallax-offset') != undefined) ? theElement.attr('parallax-offset') : 0; theElement.css('background-position', '10px ' + ((scrollPoint * factor) - offset) + 'px') }); });
